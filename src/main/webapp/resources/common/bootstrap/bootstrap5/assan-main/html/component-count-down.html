            var $this = $(this), finalDate = $(this).data('countdown');
            $this.countdown(finalDate, function (event) {
                var $this = $(this).html(event.strftime(''
                    + '<div class="width-10x height-10x d-flex flex-column flex-center rounded-circle bg-light small me-3"><h2 class="mb-0">%w</h2> weeks</div> '

                    + '<div class="width-10x height-10x d-flex flex-column flex-center rounded-circle bg-light small me-3"><h2 class="mb-0">%M</h2> min</div> '
                    + '<div class="width-10x height-10x d-flex flex-column flex-center rounded-circle bg-light small me-3"><h2 class="mb-0">%S</h2> sec</div>'));
            });
        });
